兩數運算

            
            #include <stdio.h>

            int main() {
                int x, y;
                char op;

                printf("x: ");
                scanf("%d", &x);
                printf("y: ");
                scanf("%d", &y);
                printf("Choose an operation (+ - * / %%): ");
                scanf(" %c", &op);

                switch(op) {
                    case '+': printf("%d", x + y);
                              break;
                    case '-': printf("%d", x - y);
                              break;
                    case '*': printf("%d", x * y);
                              break;
                    case '/': printf("%f\n", (float)x / y);
                              printf("%d", x / y);
                              break;
                    case '%': printf("%d", x % y);
                              break;
                    default: printf("Wrong operation!");
                }
                printf("\n");
                return 0;
            }

                        printf("%d is larger than %d.\n", n, prev);

帳戶金額存提

            
            #include <stdio.h>

            int main() {
                int opt;
                int money;
                int balance = 0;

                do {
                    printf("| 1.Withdraw | 2.Deposit | 9.Exit |\n> ");
                    scanf("%d", &opt);
                    if (opt == 9) break;
                    if (opt == 1 || opt == 2) {
                        printf("Amount: ");
                        scanf("%d", &money);
                        if (opt == 1) {
                            if (balance < money)
                                printf("The balance is not enough.\n");
                            else
                                balance -= money;
                        }
                        else {
                            balance += money;
                        }
                        printf("The balance is %d.\n", balance);
                    }
                } while (opt != 1 || opt != 2);

                return 0;
            }
