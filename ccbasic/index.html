常用算術運算

            
            #include <iostream>
            using namespace std;

            int main() {
                int a, b;

                cout << "a: ";
                cin >> a;
                cout << "b: ";
                cin >> b;
                // 加法(+)
                cout << a << " + " << b << " = " << a + b << endl;
                // 減法(-)
                cout << a << " - " << b << " = " << a - b << endl;
                // 乘法(*)
                cout << a << " * " << b << " = " << a * b << endl;
                // 除法(/) 其中之一為浮點數，結果為浮點數
                cout << a << " / " << b << " = " << (float)a / b << endl;
                // 商數(/) 兩整數相除結果為商數
                cout << a << " / " << b << " = " << a / b << endl;
                // 餘數(%)
                cout << a << " % " << b << " = " << a % b << endl;
                return 0;
            }

            #include <bits/stdc++.h>
            using namespace std;

            int main() {
                char c;
                int uppers = 0;
                int lowers = 0;

                while (cin >> c) {
                    if (c >= 'A' && c <= 'Z') {
                        uppers += 1;
                    }
                    else if (c >= 'a' && c <= 'z') {
                        lowers += 1;
                    }
                }
                cout << uppers << endl << lowers << endl;
                return 0;
            }
