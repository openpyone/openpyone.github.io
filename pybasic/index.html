Output (輸出)

基本輸出

            
            # print(要輸出的內容)
            # 參數可放任何型態的資料

            # print(1個參數)
            print('Hello, World!')
            # print(0個參數)
            print()
            # print(2個以上參數，逗號分隔)
            print('Hello, World!', 'I am R987.')
            print('哈囉！我是 R987。')

輸出單行字串

            
            # 單引號 (字串單引號 vs. 英文單引號)
            print('What\'s going on?')
            # 雙引號
            print("What's going on?")

輸出不同類別

            
            # 字串 str (string)
            print('123', type('123'))
            # 整數 int (integer)
            print(123, type(123))
            # 浮點數 float
            print(123.0, type(123.0))

            # input(提示訊息)

            # input(0個參數)

            input('What is your name? ')

            input('1.Create 2.Read 3.Update 4.Delete: ')

直式選單提示

            
            print('L.Left')

            print('U.Up')

            input('> ')

Math (計算)

算術運算

            
            """
            # ex(+). print-eqs
            # 輸出運算結果
            # 請為所有算式設定 print() 函式
            """

            # 加(+)、減(-)、乘(*)、除(/)
            7 + 2
            7 - 2
            7 * 2
            7 / 2
            # 商數(//)、餘數(%)
            7 // 2
            7 % 2
            # 指數 (次方)
            7 ** 2

運算子優先序 (Operator Precedence)

            
            # 先乘除後加減
            a = 7 + 2 - 3 * 2
            print('a:', a)
            b = 8 + 4 / 2 - 5 * 2
            print('b:', b)
            
        
            
            # 指乘除餘加減
            a = 8 * 2 + 7 - 9 // 4 * 3 ** 2
            print('a:', a)
            b = 27 % 5 * 6 ** 2 - 12 // 6
            print('b:', b)

兩數相加

            
            # 輸入兩個整數字串存到變數 a 和 b
            a = input('a: ')
            b = input('b: ')

            # 因為任何輸入都是字串
            # 所以必須使用 int() 轉成整數
            a = int(a)
            b = int(b)
            print('sum:', a+b)

程式錯誤種類

            
            """
            # 題意：將 a 和 b 以算術加法合計

            1. 語法錯誤 Syntax Error  : 不可執行
            2. 執行錯誤 Runtime Error : 可以執行，過程出錯
            3. 邏輯錯誤 Logic Error   : 可以執行，不會出錯，結果不符預期
            4. 語意錯誤 Semantic Error: 即邏輯錯誤

            type(data): 檢查資料的類別
            int('987'): 將數字轉成整數 -> integer
            str(12345): 將整數轉成數字 -> string
            """

            a = input('Input a number: ))
            b = 123
            print('type of a:', type(a))
            print('type of b:', type(b))
            print(a + b)

Decision making (決策) - 決策式(分支或選擇結構)

if 條件式

            
            print("Bala: Let's hang out.\n")
            weather = 'sunny'
            if weather == 'sunny':
                print("Sure. Why not?")

if-else 條件-否則式

            
            print("Bala: Let's hang out.\n")
            weather = input('How is the weather? ')
            if weather == 'sunny':
                print("Sure. Why not?")
            else:
                print("I want to stay at home.")

兩層巢狀決策

            
            furry = True
            small = True
            if furry:
                if small:
                    print("It's a cat!")
                else:
                    print("It's a bear!")
            else:
                if small:
                    print("It's a skink!")
                else:
                    print("It's a crocodile!")

三層巢狀決策

            
            """
            # ex(+). third-born
            # 三層巢狀決策
            # 判斷第三個孩子的性別
            """

            first_born  = 'M'
            second_born = 'F'

            if first_born == 'M':
                print('1st child: Son')
                if second_born == 'M':
                    print('2nd child: Son')
                else:
                    print('2nd child: Daughter')
            else:
                print('1st child: Daughter')
                if second_born == 'M':
                    print('2nd child: Son')
                else:
                    print('2nd child: Daughter')

Repetition (重複) - for loop 迴圈式(迴圈或重複結構)

輸出指定符號

            
            for i in range(5):
                print('*')

輸出指定數列

            
            for i in range(5):
                print(i)

迴圈與決策

            
            """
            # ex(+). diff-stars
            # 不同星號數
            # i 等於 3 時，輸出 ***，其他值則輸出 *

            1 *
            2 *
            3 ***
            4 *
            5 *
            """

            # range(stop): 0 ~ stop-1
            # range(start, stop): start ~ stop-1
            for i in range(1, 6):
                if False:
                    print(i, '***')
                else:
                    print(i, '*')

星號三角

            
            """
            # ex(+). triangles
            # 正左直角
            1 *
            2 **
            3 ***
            4 ****
            5 *****

            # 倒左直角
            1 *****
            2 ****
            3 ***
            4 **
            5 *

            # 正右直角
            1     *
            2    **
            3   ***
            4  ****
            5 *****

            # 倒右直角
            1 *****
            2  ****
            3   ***
            4    **
            5     *

            # 金字塔形
            1     *
            2    ***
            3   *****
            4  *******
            5 *********

            for i in range(1, 6):
                print(i, '*'*i)

偶數相加

            
            n = 10
            total = 0
            for i in range(1, n+1):
                if i % 2 == 0:
                    total = total + i
            print('Sum of even numbers:', total)

等差數列

            
            """
            range(stop): 0 ~ stop-1
            range(start, stop): start ~ stop-1
            range(start, stop, step): start ~ stop-1, every step
            """

            for i in range(1, 10, 2):
                print(i)
            print()

            for i in range(10, 1, 2):
                print(i)
            print()

            for i in range(10, 1, -2):
                print(i)
            print()

區分作用命名

            
            """
            # ex(+). same-for
            # 區分作用命名
            # 使兩個相同迴圈輸出同樣結果
            # 不得增加程式行數
            # range() 參數不得直接放整數或算式，但可放變數
            """

            i = 5
            for i in range(i):
                print('i:', i)
                i = i * 2
                print('Double of i =', i)

            print('-'*20)

            for i in range(i):
                print('i:', i)
                i = i * 2
                print('Double of i =', i)

Repetition (重複) - while loop 迴圈式(迴圈或重複結構)

while 布林變數

            
            isplus = True
            total = 0
            while isplus:
                total = total + 1
                print(total)
                if total == 10:
                    isplus = False

while 布林值

            
            total = 0
            while True:
                total += 1
                print(total)
                if total == 10:
                    break

限定等差數列

            
            """
            # ex(+). arithmetic-seq
            # 限定等差數列
            # 維持使用一個條件式
            # 條件與流程都需要調整
            # 比較雙方維持 total 和 10

            # 比較運算子
            == 等於     != 不等於
            >= 大於等於  > 大於
            <= 小於等於  < 小於

            [輸入與輸出]
            step: 2
            2
            4
            6
            8
            10

            step: 3
            3
            6
            9

            step: 4
            4
            8
            """

            total = 0
            while True:
                total += 1
                print(total)
                if total == 10:
                    break

while 條件式

            
            """
            # ex(+). while-condition
            # 條件迴圈式
            # while 依條件式決定是否執行
            # 程式行數必須減至 4 行
            """

            total = 0
            while False:
                total += 1
                print(total)
                if total == 10:
                    break

            """

            # 程式行數必須均為 4 行
